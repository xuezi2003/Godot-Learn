# 07 敌人系统实现（下）

## 脚本结构

```gdscript
class_name Enemy
extends CharacterBody2D

const DEFAULT_BULLET_DAMAGE := 1
const EXPLOSION_QUERY_MAX_RESULTS := 32

enum DeathSequenceStage {
    NONE,
    DYING,
    EXPLODING
}

@export var config: EnemyConfig
@export var touch_damage: int = 1
@export var touch_damage_interval: float = 0.5
@export var hit_blink_duration: float = 0.1

@onready var sprite: AnimatedSprite2D = $AnimatedSprite2D
@onready var body_shape: CollisionShape2D = $CollisionShape2D
@onready var touch_damage_area: Area2D = $TouchDamageArea
@onready var touch_damage_shape: CollisionShape2D = $TouchDamageArea/CollisionShape2D
@onready var explosion_area: Area2D = $ExplosionArea
@onready var explosion_shape: CollisionShape2D = $ExplosionArea/CollisionShape2D

var target_player: Player
var current_health: int
var is_dead := false
var touch_damage_cooldown_left := 0.0
var touching_player: Player
var hit_blink_time_left := 0.0
var death_sequence_stage := DeathSequenceStage.NONE
var death_animation_name_in_use := ""
var random_generator := RandomNumberGenerator.new()
```

## 初始化

```gdscript
func _ready() -> void:
    random_generator.randomize()
    touch_damage_area.body_entered.connect(_on_touch_damage_area_body_entered)
    touch_damage_area.body_exited.connect(_on_touch_damage_area_body_exited)
    touch_damage_area.area_entered.connect(_on_touch_damage_area_area_entered)
    sprite.animation_finished.connect(_on_animation_finished)
    apply_config()

func setup(enemy_config: EnemyConfig, player: Player) -> void:
    config = enemy_config
    target_player = player
    apply_config()

func set_target(player: Player) -> void:
    target_player = player
```

## 应用配置

```gdscript
func apply_config() -> void:
    if config == null:
        return

    current_health = config.max_health
    apply_collision_radius(config.collision_radius)
    apply_explosion_radius(config.explosion_radius)

    if config.sprite_frames != null:
        sprite.sprite_frames = config.sprite_frames
        sprite.play(config.move_animation_name)

func apply_collision_radius(radius: float) -> void:
    body_shape.shape.radius = radius
    touch_damage_shape.shape.radius = radius + 2.0

func apply_explosion_radius(radius: float) -> void:
    explosion_shape.shape.radius = maxf(radius, 0.0)

func get_move_speed() -> float:
    if config == null:
        return 0.0
    return config.move_speed
```

## 物理更新

```gdscript
func _physics_process(delta: float) -> void:
    update_hit_blink(delta)
    update_touch_damage(delta)

    if is_dead:
        velocity = Vector2.ZERO
        return

    if not is_instance_valid(target_player):
        velocity = Vector2.ZERO
        return

    var direction := global_position.direction_to(target_player.global_position)
    update_facing(direction)
    velocity = direction * get_move_speed()
    move_and_slide()
```

- 敌人直接朝玩家直线移动，不引入复杂寻路。
- `move_and_slide()` 会自动处理与障碍物的滑动，避免完全卡死。

## 更新朝向

```gdscript
func update_facing(direction: Vector2) -> void:
    if not is_zero_approx(direction.x):
        sprite.flip_h = direction.x < 0
```

## 受伤与死亡

```gdscript
func apply_damage(damage: int) -> void:
    if is_dead or damage <= 0:
        return

    current_health -= damage
    if current_health <= 0:
        die()
    else:
        start_hit_blink()

func die() -> void:
    if is_dead:
        return

    is_dead = true
    velocity = Vector2.ZERO
    touching_player = null
    set_hit_blink_enabled(false)

    body_shape.set_deferred("disabled", true)
    touch_damage_area.set_deferred("monitoring", false)
    touch_damage_area.set_deferred("monitorable", false)

    _try_drop_pickup()
    start_death_sequence()
```

- 使用 `set_deferred` 延迟禁用碰撞形状和 Area2D 监控，避免在物理回调中直接修改导致报错。

## 接触伤害

```gdscript
func _on_touch_damage_area_body_entered(body: Node2D) -> void:
    if is_dead:
        return
    if body is Player:
        touching_player = body
        _try_deal_touch_damage()

func _on_touch_damage_area_body_exited(body: Node2D) -> void:
    if body == touching_player:
        touching_player = null

func update_touch_damage(delta: float) -> void:
    if touch_damage_cooldown_left > 0.0:
        touch_damage_cooldown_left -= delta
    _try_deal_touch_damage()

func _try_deal_touch_damage() -> void:
    if touching_player == null or not is_instance_valid(touching_player):
        return
    if touch_damage_cooldown_left > 0.0:
        return

    touching_player.apply_damage(touch_damage)
    touch_damage_cooldown_left = touch_damage_interval
```

## 子弹命中

```gdscript
func _on_touch_damage_area_area_entered(area: Area2D) -> void:
    if is_dead:
        return
    if area is Bullet:
        apply_damage(DEFAULT_BULLET_DAMAGE)
        area.queue_free()
```

- 子弹命中由敌人自身处理，便于扩展不同敌人的受击表现。

## 受击闪烁

```gdscript
func start_hit_blink() -> void:
    hit_blink_time_left = hit_blink_duration
    set_hit_blink_enabled(true)

func update_hit_blink(delta: float) -> void:
    if hit_blink_time_left > 0.0:
        hit_blink_time_left -= delta
        if hit_blink_time_left <= 0.0:
            set_hit_blink_enabled(false)

func set_hit_blink_enabled(enabled: bool) -> void:
    var material := sprite.material as ShaderMaterial
    if material != null:
        material.set_shader_parameter("blink_enabled", enabled)
```

## 死亡动画流程

```gdscript
func start_death_sequence() -> void:
    if config == null:
        queue_free()
        return
    play_death_sequence_animation(config.death_animation_name, DeathSequenceStage.DYING)

func _finish_after_death_animation() -> void:
    if should_play_explosion_sequence():
        start_explosion_sequence()
    else:
        queue_free()

func start_explosion_sequence() -> void:
    _try_apply_explosion_damage()
    play_death_sequence_animation(&"explode", DeathSequenceStage.EXPLODING)

func play_death_sequence_animation(animation_name: StringName, stage: DeathSequenceStage) -> void:
    if config == null or config.sprite_frames == null:
        queue_free()
        return
    if not config.sprite_frames.has_animation(animation_name):
        queue_free()
        return

    death_sequence_stage = stage
    death_animation_name_in_use = animation_name
    sprite.play(animation_name)

func should_play_explosion_sequence() -> bool:
    return config != null and config.explode_on_death
```

## 爆炸伤害

```gdscript
func _try_apply_explosion_damage() -> void:
    if config == null or not config.explode_on_death:
        return
    if config.explosion_damage <= 0 or config.explosion_radius <= 0.0:
        return
    if explosion_shape == null:
        return

    var space_state := get_world_2d().direct_space_state
    var query := PhysicsShapeQueryParameters2D.new()
    query.shape = explosion_shape.shape
    query.transform = explosion_area.global_transform
    query.collision_mask = explosion_area.collision_mask

    var results := space_state.intersect_shape(query, EXPLOSION_QUERY_MAX_RESULTS)
    var damaged_collider_ids: Array[int] = []

    for result in results:
        var collider_id: int = result["collider_id"]
        if collider_id in damaged_collider_ids:
            continue
        damaged_collider_ids.append(collider_id)

        var collider: Node2D = result["collider"]
        if collider is Player or collider is Enemy:
            collider.apply_damage(config.explosion_damage)
```

- 使用 `intersect_shape` 主动查询爆炸范围内的重叠对象。
- 用 `collider_id` 去重，避免同一对象多个碰撞形状被多次伤害。

## 掉落道具

```gdscript
func _try_drop_pickup() -> void:
    if config == null or config.possible_drops.is_empty():
        return
    if random_generator.randf() > config.drop_chance:
        return

    var drop_config := _pick_pickup_drop_config()
    if drop_config != null:
        call_deferred("_spawn_dropped_pickup", drop_config)

func _pick_pickup_drop_config() -> PickUpConfig:
    if config == null:
        return null

    var valid_drops: Array[PickUpConfig] = []
    var total_weight := 0
    for drop in config.possible_drops:
        if drop == null or drop.drop_weight <= 0:
            continue
        valid_drops.append(drop)
        total_weight += drop.drop_weight

    if valid_drops.is_empty():
        return null

    var roll := random_generator.randi_range(1, total_weight)
    var accumulated := 0
    for drop in valid_drops:
        accumulated += drop.drop_weight
        if roll <= accumulated:
            return drop

    return valid_drops.back()

func _spawn_dropped_pickup(drop_config: PickUpConfig) -> void:
    var pickup := preload("res://scenes/pick_up.tscn").instantiate() as PickUp
    pickup.config = drop_config
    get_tree().current_scene.add_child(pickup)
    pickup.global_position = global_position
```

- 按权重随机选择掉落物。
- 使用 `call_deferred` 延迟生成道具，避免在物理回调中直接创建节点。

## 动画结束处理

```gdscript
func _on_animation_finished() -> void:
    if not is_dead:
        return
    if death_animation_name_in_use.is_empty():
        return
    if sprite.animation != death_animation_name_in_use:
        return

    match death_sequence_stage:
        DeathSequenceStage.DYING:
            _finish_after_death_animation()
        DeathSequenceStage.EXPLODING:
            queue_free()
```
