# 08 敌人刷新与战斗闭环

## 游戏场景结构整理

1. 在 `game` 场景中创建 **Node2D**，命名为 `EnemyContainer`。
   - 所有生成的敌人都作为该节点的子节点，便于统一管理和计数。
2. 创建 **Node2D**，命名为 `EnemySpawnPoints`。
3. 在其下添加 **Marker2D** 作为出生点：
   - `SpawnLeft`：地图左侧入口
   - `SpawnRight`：地图右侧入口
   - `SpawnBottom`：地图底部入口
4. 添加 **Timer**，命名为 `EnemySpawnTimer`，控制刷新节奏。
5. 将 `WorldBoundaries` 拖到场景树末尾，按逻辑关系分组。

> Marker2D 是通用的 2D 位置标记节点，适合出生点、目标点、路径点、相机参考点等。也可以用 Node2D 替代，但 Marker2D 语义更清晰。

## Game 脚本：敌人刷新

### 核心设计

- 游戏目标：生存到倒计时结束。
- 难度线性提升：敌人属性不变，但刷新间隔逐渐缩短，敌人数增多。
- 限制最大刷新频率和场上最大敌人数，防止后期失控。

### 常量与变量

```gdscript
const ENEMY_SCENE := preload("res://scenes/enemy.tscn")

@export var start_spawn_interval: float = 1.5
@export var end_spawn_interval: float = 0.6
@export var spawn_acceleration_duration: float = 60.0
@export var initial_spawn_count: int = 3
@export var min_spawn_interval: float = 0.1
@export var max_alive_enemies: int = 50

@onready var enemy_container: Node2D = $EnemyContainer
@onready var enemy_spawn_points: Node2D = $EnemySpawnPoints
@onready var enemy_spawn_timer: Timer = $EnemySpawnTimer
@onready var player: Player = $Player

var enemy_spawn_positions: Array[Marker2D] = []
var enemy_configs: Array[EnemyConfig] = []
var game_time := 0.0
```

### 初始化

```gdscript
func _ready() -> void:
    scan_spawn_points()
    cache_enemy_configs()
    setup_enemy_spawn_timer()
    spawn_initial_enemies()

func _process(delta: float) -> void:
    game_time += delta
    update_spawn_interval()
```

### 缓存出生点与配置

```gdscript
func scan_spawn_points() -> void:
    enemy_spawn_positions.clear()
    for child in enemy_spawn_points.get_children():
        if child is Marker2D:
            enemy_spawn_positions.append(child)

func cache_enemy_configs() -> void:
    enemy_configs.clear()
    var config_files := DirAccess.get_files_at("res://resources/config")
    for file in config_files:
        if file.ends_with(".tres"):
            var config := load("res://resources/config/" + file) as EnemyConfig
            if config != null:
                enemy_configs.append(config)
```

### 刷新计时器

```gdscript
func setup_enemy_spawn_timer() -> void:
    enemy_spawn_timer.wait_time = get_current_spawn_interval()
    enemy_spawn_timer.timeout.connect(_on_enemy_spawn_timer_timeout)
    enemy_spawn_timer.start()

func update_spawn_interval() -> void:
    var new_interval := get_current_spawn_interval()
    enemy_spawn_timer.wait_time = new_interval
    enemy_spawn_timer.start(new_interval)

func get_current_spawn_interval() -> float:
    var start := maxf(start_spawn_interval, 0.1)
    var end := clampf(end_spawn_interval, 0.1, start)
    if spawn_acceleration_duration <= 0.0:
        return end

    var ratio := clampf(game_time / spawn_acceleration_duration, 0.0, 1.0)
    return lerp(start, end, ratio)
```

- 调用 `start()` 重新启动计时器，使新的间隔立即生效。

### 初始敌人与定时刷新

```gdscript
func spawn_initial_enemies() -> void:
    for i in initial_spawn_count:
        try_spawn_enemy()

func _on_enemy_spawn_timer_timeout() -> void:
    try_spawn_enemy()

func try_spawn_enemy() -> bool:
    if not is_spawn_system_ready():
        return false
    if get_alive_enemy_count() >= max_alive_enemies:
        return false

    var spawn_point := pick_spawn_point()
    var config := pick_enemy_config()
    if spawn_point == null or config == null:
        return false

    var enemy := ENEMY_SCENE.instantiate() as Enemy
    enemy_container.add_child(enemy)
    enemy.global_position = spawn_point.global_position
    enemy.setup(config, player)
    return true
```

### 辅助函数

```gdscript
func is_spawn_system_ready() -> bool:
    return player != null and ENEMY_SCENE != null and not enemy_spawn_positions.is_empty() and not enemy_configs.is_empty()

func pick_spawn_point() -> Marker2D:
    if enemy_spawn_positions.is_empty():
        return null
    return enemy_spawn_positions[randi() % enemy_spawn_positions.size()]

func pick_enemy_config() -> EnemyConfig:
    if enemy_configs.is_empty():
        return null
    return enemy_configs[randi() % enemy_configs.size()]

func get_alive_enemy_count() -> int:
    var count := 0
    for child in enemy_container.get_children():
        if child is Enemy and not child.is_dead:
            count += 1
    return count
```

## Player 脚本更新

### 新增常量与变量

```gdscript
const BLINK_ENABLED_PARAM := "blink_enabled"

@export var max_health: int = 3
@export var invincibility_duration: float = 0.5

var current_health: int
var invincibility_time_left := 0.0
var is_dead := false
```

### 初始化

```gdscript
func _ready() -> void:
    current_health = max_health
    set_hit_blink_enabled(false)
    # ... 其他初始化
```

### 物理更新

```gdscript
func _physics_process(delta: float) -> void:
    update_invincibility(delta)

    if is_dead:
        velocity = Vector2.ZERO
        return

    # ... 移动、射击等逻辑
```

### 受伤与无敌

```gdscript
func apply_damage(damage: int) -> bool:
    if is_dead or damage <= 0 or invincibility_time_left > 0.0:
        return false

    current_health -= damage
    if current_health <= 0:
        _die()
    else:
        start_invincibility()
    return true

func get_current_health() -> int:
    return current_health

func update_invincibility(delta: float) -> void:
    if invincibility_time_left > 0.0:
        invincibility_time_left -= delta
        if invincibility_time_left <= 0.0:
            set_hit_blink_enabled(false)

func start_invincibility() -> void:
    invincibility_time_left = invincibility_duration
    set_hit_blink_enabled(true)

func set_hit_blink_enabled(enabled: bool) -> void:
    var material := body_sprite.material as ShaderMaterial
    if material != null:
        material.set_shader_parameter(BLINK_ENABLED_PARAM, enabled)

func _die() -> void:
    is_dead = true
    velocity = Vector2.ZERO
    set_hit_blink_enabled(false)
    is_armed_mode = false
    is_spiral_mode = false
```

### 玩家 BodySprite 应用闪烁 Shader

1. 选中 `BodySprite`。
2. 在 **CanvasItem > Material** 中新建 **ShaderMaterial**。
3. 加载 `blink.gdshader`。
4. 只有一个玩家角色，是否勾选 Local to Scene 影响不大。

## 修复敌人死亡动画循环问题

1. 打开每个敌人的 SpriteFrames 资源。
2. 关闭 `death` 动画的 **Loop**。
3. 将 `death` 动画速度从 5 FPS 改为 10 FPS。
4. 自爆敌人的 `explode` 动画同样关闭 Loop。

## 修复子弹贴墙生成穿墙问题

### 问题原因

子弹从玩家位置向射击方向偏移后生成。当玩家紧贴墙壁或地图边界时，子弹出生点可能已经在障碍物内部或地图外，导致之前的碰撞检测失效。

### 解决方案

在生成子弹前，用射线检测玩家中心到子弹出生点之间是否存在障碍物。

```gdscript
func _can_spawn_bullet(spawn_position: Vector2) -> bool:
    var query := PhysicsRayQueryParameters2D.create(global_position, spawn_position, Bullet.WORLD_COLLISION_MASK)
    var result := get_world_2d().direct_space_state.intersect_ray(query)
    return result.is_empty()

func spawn_bullet(direction: Vector2) -> bool:
    var spawn_position := global_position + direction * bullet_spawn_distance
    if not _can_spawn_bullet(spawn_position):
        return false

    var bullet := BULLET_SCENE.instantiate() as Bullet
    get_tree().current_scene.add_child(bullet)
    bullet.global_position = spawn_position
    bullet.setup(direction)
    return true
```
